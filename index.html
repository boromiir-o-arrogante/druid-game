<!DOCTYPE html>

<html lang="pt-BR">

<head>

  <meta charset="UTF-8">

  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <title>Druid's Descent</title>

  <script src="https://cdn.tailwindcss.com"></script>

  <script src="https://unpkg.com/react@18/umd/react.development.js" crossorigin></script>

  <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js" crossorigin></script>

  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

</head>

<body class="m-0 bg-slate-950 overflow-hidden">

  <div id="root"></div>

  <script type="text/babel">

    const { useState, useEffect, useCallback } = React;


    const EMOJIS = { wall: "🌲", floor: "·", player: "🧙‍♂️", goblin: "👹", slime: "🦠", orc: "🧌", bat: "🦇", fog: "⬛" };

    const ENEMY_TYPES = [

      { type: 'bat', hp: 2, xp: 1, name: 'Morcego' }, { type: 'slime', hp: 2, xp: 1, name: 'Slime' },

      { type: 'goblin', hp: 4, xp: 2, name: 'Goblin' }, { type: 'orc', hp: 8, xp: 4, name: 'Orc' }

    ];


    const MAP_WIDTH = 20; const MAP_HEIGHT = 15; const OBSTACLE_CHANCE = 0.15; const MAX_ENEMIES = 10; const VISION_RADIUS = 5;


    const isVisible = (cellX, cellY, player) => {

      const dx = cellX - player.x; const dy = cellY - player.y;

      const absDx = Math.abs(dx); const absDy = Math.abs(dy);

      if (absDx <= 1 && absDy <= 1) return true; 

      if (absDx > VISION_RADIUS || absDy > VISION_RADIUS) return false;

      if (player.facing === 'UP' && dy < 0 && absDx <= Math.abs(dy)) return true;

      if (player.facing === 'DOWN' && dy > 0 && absDx <= dy) return true;

      if (player.facing === 'LEFT' && dx < 0 && absDy <= Math.abs(dx)) return true;

      if (player.facing === 'RIGHT' && dx > 0 && absDy <= dx) return true;

      return false;

    };


    const generateMap = (width, height, startX, startY) => {

      const newMap = [];

      for (let y = 0; y < height; y++) {

        const row = [];

        for (let x = 0; x < width; x++) {

          const isBorder = y === 0 || y === height - 1 || x === 0 || x === width - 1;

          const isSpawnPoint = x === startX && y === startY;

          const isInternalWall = Math.random() < OBSTACLE_CHANCE && !isSpawnPoint && !isBorder;

          row.push({ x, y, terrain: isBorder || isInternalWall ? 'wall' : 'floor', entity: null, explored: false });

        }

        newMap.push(row);

      }

      let enemiesSpawned = 0;

      while (enemiesSpawned < MAX_ENEMIES) {

        const randomX = Math.floor(Math.random() * width); const randomY = Math.floor(Math.random() * height);

        const targetCell = newMap[randomY][randomX]; const isSpawnPoint = randomX === startX && randomY === startY;

        if (targetCell.terrain === 'floor' && !targetCell.entity && !isSpawnPoint) {

          targetCell.entity = { ...ENEMY_TYPES[Math.floor(Math.random() * ENEMY_TYPES.length)] };

          enemiesSpawned++;

        }

      }

      return newMap;

    };


    function DruidGame() {

      const [gameState, setGameState] = useState(() => {

        const startX = Math.floor(MAP_WIDTH / 2); const startY = Math.floor(MAP_HEIGHT / 2);

        const initialMap = generateMap(MAP_WIDTH, MAP_HEIGHT, startX, startY);

        const initialPlayer = { x: startX, y: startY, hp: 15, maxHp: 15, damage: 2, facing: 'DOWN', level: 1, xp: 0, statPoints: 0 };

        initialMap[startY][startX].entity = { type: 'player' };

        initialMap.forEach(row => row.forEach(cell => { if (isVisible(cell.x, cell.y, initialPlayer)) cell.explored = true; }));

        return { map: initialMap, player: initialPlayer, log: ["Bem-vindo! Olhe ao redor com WASD."] };

      });


      const allocateStat = (stat) => {

        setGameState(prev => {

          const newPlayer = { ...prev.player };

          if (stat === 'hp') { newPlayer.maxHp += 5; newPlayer.hp += 5; } else if (stat === 'damage') { newPlayer.damage += 1; }

          newPlayer.statPoints -= 1;

          return { ...prev, player: newPlayer };

        });

      };


      const movePlayer = useCallback((dx, dy) => {

        setGameState((prev) => {

          const { map, player, log } = prev;

          if (player.statPoints > 0) return prev; 

          let newFacing = player.facing;

          if (dx === 0 && dy === -1) newFacing = 'UP'; else if (dx === 0 && dy === 1) newFacing = 'DOWN';

          else if (dx === -1 && dy === 0) newFacing = 'LEFT'; else if (dx === 1 && dy === 0) newFacing = 'RIGHT';

          const newPlayer = { ...player, facing: newFacing };

          const newX = player.x + dx; const newY = player.y + dy;

          if (newY < 0 || newY >= MAP_HEIGHT || newX < 0 || newX >= MAP_WIDTH) return prev;

          const targetCell = map[newY][newX]; const newMap = map.map(row => [...row]); let newLog = [...log]; 

          const updateVision = (p) => { newMap.forEach(row => row.forEach(cell => { if (isVisible(cell.x, cell.y, p)) cell.explored = true; })); };


          if (targetCell.terrain === 'wall') { updateVision(newPlayer); return { ...prev, map: newMap, player: newPlayer }; }


          if (targetCell.entity && targetCell.entity.type !== 'player') {

            const enemy = { ...targetCell.entity }; enemy.hp -= newPlayer.damage;

            newLog.unshift(`Você atacou o ${enemy.name} (${newPlayer.damage} dano).`);

            if (enemy.hp <= 0) {

              newLog.unshift(`O ${enemy.name} morreu! (+${enemy.xp} XP)`);

              newMap[newY][newX] = { ...targetCell, entity: null };

              newPlayer.xp += enemy.xp; const xpNeeded = newPlayer.level * 5; 

              if (newPlayer.xp >= xpNeeded) {

                newPlayer.level += 1; newPlayer.xp -= xpNeeded; newPlayer.statPoints += 1;

                newLog.unshift(`🌟 LEVEL UP! Você chegou ao Nível ${newPlayer.level}!`);

              }

            } else { newMap[newY][newX] = { ...targetCell, entity: enemy }; }

            updateVision(newPlayer); return { map: newMap, player: newPlayer, log: newLog.slice(0, 6) };

          }


          newMap[player.y][player.x] = { ...newMap[player.y][player.x], entity: null };

          newMap[newY][newX] = { ...newMap[newY][newX], entity: { type: 'player' } };

          newPlayer.x = newX; newPlayer.y = newY;

          updateVision(newPlayer); return { map: newMap, player: newPlayer, log: newLog };

        });

      }, []);


      useEffect(() => {

        const handleKeyDown = (e) => {

          if (gameState.player.statPoints > 0) return;

          if(["ArrowUp","ArrowDown","ArrowLeft","ArrowRight"].includes(e.code)) e.preventDefault();

          switch (e.key) {

            case 'ArrowUp': case 'w': case 'W': movePlayer(0, -1); break;

            case 'ArrowDown': case 's': case 'S': movePlayer(0, 1); break;

            case 'ArrowLeft': case 'a': case 'A': movePlayer(-1, 0); break;

            case 'ArrowRight': case 'd': case 'D': movePlayer(1, 0); break;

          }

        };

        window.addEventListener('keydown', handleKeyDown);

        return () => window.removeEventListener('keydown', handleKeyDown);

      }, [movePlayer, gameState.player.statPoints]);


      return (

        <div className="relative flex flex-col items-center justify-center min-h-screen bg-slate-950 text-slate-200 font-mono p-4">

          {gameState.player.statPoints > 0 && (

            <div className="absolute inset-0 bg-black/80 flex items-center justify-center z-50">

              <div className="bg-slate-900 border-2 border-amber-500 p-6 rounded-xl text-center shadow-[0_0_30px_rgba(245,158,11,0.3)] max-w-sm">

                <h2 className="text-3xl text-amber-400 mb-2 font-bold">LEVEL UP!</h2>

                <p className="text-slate-300 mb-6">Você tem {gameState.player.statPoints} ponto(s).</p>

                <div className="flex flex-col gap-3">

                  <button onClick={() => allocateStat('hp')} className="bg-slate-800 border border-emerald-500 text-emerald-400 py-3 px-4 rounded">+5 HP Máximo</button>

                  <button onClick={() => allocateStat('damage')} className="bg-slate-800 border border-red-500 text-red-400 py-3 px-4 rounded">+1 Dano</button>

                </div>

              </div>

            </div>

          )}

          <h1 className="text-emerald-500 text-3xl mb-4 font-bold tracking-widest">DRUID's DESCENT</h1>

          <div className="flex flex-col xl:flex-row gap-6">

            <div className="border-4 border-slate-700 bg-black p-2 rounded-lg shadow-2xl">

              {gameState.map.map((row, y) => (

                <div key={y} className="flex">

                  {row.map((cell, x) => {

                    const isCurrentlyVisible = isVisible(x, y, gameState.player); const isExplored = cell.explored;

                    let content = "⬛"; let extraClass = "opacity-0"; 

                    if (isCurrentlyVisible) { content = cell.entity ? EMOJIS[cell.entity.type] : EMOJIS[cell.terrain]; extraClass = "opacity-100"; } 

                    else if (isExplored) { content = EMOJIS[cell.terrain]; extraClass = "opacity-30 grayscale"; }

                    return <div key={`${x}-${y}`} className={`w-8 h-8 md:w-10 md:h-10 flex items-center justify-center text-xl md:text-2xl select-none transition-all duration-300 ${extraClass}`}>{content}</div>;

                  })}

                </div>

              ))}

            </div>

            <div className="flex flex-col w-full xl:w-80 gap-4">

              <div className="bg-slate-900 border-2 border-slate-700 p-4 rounded-lg">

                <h2 className="text-emerald-400 font-bold mb-2 uppercase border-b border-slate-700 pb-1">Status</h2>

                <div className="flex justify-between items-center text-lg mb-1">

                  <span>{EMOJIS.player} Nvl. {gameState.player.level}</span>

                  <span className="text-red-400 font-bold">HP: {gameState.player.hp}/{gameState.player.maxHp}</span>

                </div>

                <div className="w-full bg-slate-800 rounded-full h-2.5 mb-2">

                  <div className="bg-blue-500 h-2.5 rounded-full transition-all" style={{ width: `${(gameState.player.xp / (gameState.player.level * 5)) * 100}%` }}></div>

                </div>

                <div className="text-sm text-slate-400 flex justify-between"><span>Dano: {gameState.player.damage}</span><span>Visão: {gameState.player.facing}</span></div>

              </div>

              <div className="bg-slate-900 border-2 border-slate-700 p-4 rounded-lg flex-1 overflow-hidden">

                <h2 className="text-slate-400 font-bold mb-2 uppercase border-b border-slate-700 pb-1">Diário</h2>

                <ul className="text-sm space-y-2">

                  {gameState.log.map((msg, index) => (<li key={index} className={`${index === 0 ? 'text-slate-100 font-semibold' : 'text-slate-500'}`}>{index === 0 ? '> ' : ''}{msg}</li>))}

                </ul>

              </div>

            </div>

          </div>

        </div>

      );

    }

    const root = ReactDOM.createRoot(document.getElementById("root"));

    root.render(<DruidGame />);

  </script>

</body>

</html>
